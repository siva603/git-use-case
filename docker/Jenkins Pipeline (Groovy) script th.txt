# Jenkins Pipeline (Groovy) — Full Automation

```groovy
pipeline {

    agent any

    tools {
        maven 'Maven'
        jdk 'JDK17'
    }

    environment {

        APP_NAME = "devops-demo"
        IMAGE_NAME = "devops-demo-image"
        CONTAINER_NAME = "devops-demo-container"
        DOCKER_PORT = "8080"

    }

    stages {

        stage('Clean Workspace') {
            steps {
                cleanWs()
            }
        }

        stage('Clone Repository') {
            steps {
                git branch: 'main',
                url: 'https://github.com/your-username/devops-maven-docker-jenkins-project.git'
            }
        }

        stage('Verify Tools') {
            steps {
                sh 'java -version'
                sh 'mvn -version'
                sh 'docker --version'
            }
        }

        stage('Maven Build') {
            steps {
                sh 'mvn clean package'
            }
        }

        stage('Build Docker Image') {
            steps {
                sh """
                docker build -t ${IMAGE_NAME}:latest .
                """
            }
        }

        stage('Stop Old Container') {
            steps {
                sh """
                docker stop ${CONTAINER_NAME} || true
                docker rm ${CONTAINER_NAME} || true
                """
            }
        }

        stage('Run Docker Container') {
            steps {
                sh """
                docker run -d \
                --name ${CONTAINER_NAME} \
                -p ${DOCKER_PORT}:8080 \
                ${IMAGE_NAME}:latest
                """
            }
        }

        stage('Verify Deployment') {
            steps {
                sh "curl http://localhost:${DOCKER_PORT} || true"
            }
        }

    }

    post {

        success {
            echo 'Pipeline completed successfully 🚀'
        }

        failure {
            echo 'Pipeline failed ❌'
        }

        always {
            sh 'docker ps'
        }

    }
}
```

# Jenkins Job Setup

### 1️⃣ Install Required Tools in Jenkins

Manage Jenkins → Global Tool Configuration

Add:

```
JDK 17
Maven 3.9+
Docker installed on server
```

---

### 2️⃣ Create Pipeline Job

Jenkins → **New Item**

Select:

```
Pipeline
```

---

### 3️⃣ Add Jenkinsfile

Pipeline → Definition:

```
Pipeline Script
```

Paste the Groovy code.
